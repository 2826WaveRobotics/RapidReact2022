
OPERATOR CONTROLS MAPPING

// Transport
right trigger = transporter / indexer
Y + right trigger = REVERSE transporter / indexer
left joystick (y) = transporter
right joystick (y) = indexer

// Intake
left trigger = intake IN 
Y + left trigger = intake OUT
select + right bumper = hook deploy
select + left bumber = hook stow

// Punch
A = both punchers
X = front/outer punch
B = back/inner punch
